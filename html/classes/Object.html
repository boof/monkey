def first(len = 1)
  raise ArgumentError, "negative string size" if len.negative?

  return if empty?
  return "" if len.zero?

  slice(...len)
end

def last(len = 1)
  raise ArgumentError, "negative string size" if len.negative?

  return if empty?
  return "" if len.zero?

  slice(-len..)
end
