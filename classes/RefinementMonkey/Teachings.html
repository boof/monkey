def read(name)
  @mutex.synchronize do
    next if seen.member? name

    monkey  = @monkey
    context = Module.new do
      define_method :patch, ->(o, &b) { monkey.patch(o, &b) }
    end

    owner_path = File.join path, "#{name}.rb"
    load owner_path, context

    seen << name
  end
end
